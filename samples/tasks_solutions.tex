\subsection*{Задача №1}
\textbf{Функция цели: }
maximize $ -1x_{1} -3x_{2} $\\

\textbf{Ограничения:}

\[ \begin{aligned}
0x_{1} -5x_{2} &\le 5 \\ 
-5x_{1} -5x_{2} &\le 2 \\ 
x_i &\ge 0,\\ 
\quad i=1,\dots,2 \\ 
\end{aligned}\]

\textbf{Оптимальное значение: }$-0.00$

\textbf{Решение: }$x_{1}=0.00, x_{2}=0.00$
